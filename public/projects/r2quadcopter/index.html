Here is a simple example of C code to blink an LED:

#include <avr/io.h>
#include <util/delay.h>

#define LED_PIN PB0

int main(void) {
    // Set LED_PIN as an output
    DDRB |= (1 << LED_PIN);

    while (1) {
        // Toggle the LED
        PORTB ^= (1 << LED_PIN);
        _delay_ms(1000); // Wait for 1 second
    }

    return 0;
}
